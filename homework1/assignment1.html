A palindrome is a word, phrase, or sequence that reads the same backward as forward. For example, both “madam” and “nurses run” are palindromes.

def is_palindrome(string):
    processed_string = ''.join(char.lower() for char in string if char.isalnum())
    
    return processed_string == processed_string[::-1]

6. You are given two non-empty linked lists representing two non-negative integers.¶

The digits are stored in reverse order and each of their nodes contain a single digit. Add the two numbers and return it as a linked list. You may assume the two numbers do not contain any leading zero, except the number 0 itself.

e.g.,

Input: (2 -> 4 -> 3) + (5 -> 6 -> 4)

Output: 7 -> 0 -> 8

In [ ]:
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def addTwoNumbers(l1, l2):
    head = ListNode(0)
    current = head
    carry = 0

    while l1 or l2:
        val1 = l1.val if l1 else 0
        val2 = l2.val if l2 else 0
        total = val1 + val2 + carry
        carry = total // 10
        remainder = total % 10

        current.next = ListNode(remainder)
        current = current.next

        if l1:
            l1 = l1.next
        if l2:
            l2 = l2.next

    if carry > 0:
        current.next = ListNode(carry)

    return head.next

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

#eg:

def merge_sort(arr):
    if len(arr) > 1:
        mid = len(arr) // 2 
        L = arr[:mid]  
        R = arr[mid:]

        merge_sort(L)  
        merge_sort(R)  
        i = j = k = 0

        while i < len(L) and j < len(R):
            if L[i] < R[j]:
                arr[k] = L[i]
                i += 1
            else:
                arr[k] = R[j]
                j += 1
            k += 1

        while i < len(L):
            arr[k] = L[i]
            i += 1
            k += 1

        while j < len(R):
            arr[k] = R[j]
            j += 1
            k += 1

    return arr

def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    else:
        pivot = arr[0]
        less = [x for x in arr[1:] if x < pivot]
        equal = [x for x in arr if x == pivot]
        greater = [x for x in arr[1:] if x > pivot]
        return quick_sort(less) + equal + quick_sort(greater)

#eg:
input = [10, 72, 80, 91, 21, 57]
output = quick_sort(input)
print(output)

[10, 21, 57, 72, 80, 91]

def shell_sort(arr):
    n = len(arr)
    gap = n // 2  

    while gap > 0:
        for i in range(gap, n):
            temp = arr[i]

            j = i
            while j >= gap and arr[j - gap] > temp:
                arr[j] = arr[j - gap]
                j -= gap

            arr[j] = temp
        gap //= 2

    return arr

#eg:

class LoadingCIFAR10ImageDataset(Dataset):
    def __init__(self, root_dir, transform=None):
        self.root_dir = root_dir
        self.transform = transform
        self.images = []
        
        for root, _, files in os.walk(root_dir):
            for file in files:
                if file.endswith(".png"):
                    self.images.append(os.path.join(root, file))
    
    def __len__(self):
        return len(self.images)
    
    def __getitem__(self, idx):
        img_path = self.images[idx]
        image = Image.open(img_path)
        
        if self.transform:
            image = self.transform(image)
        
        # 假设label在文件名字里
        target = img_path.split('/')[-1].split('_')[0] 
        
        return image, target

class SymmetricArray(np.ndarray):
    def __new__(cls, input_array):
        obj = np.asarray(input_array).view(cls)
        return obj

    def __getitem__(self, key):
        i, j = key
        return super().__getitem__((i, j)) if i <= j else super().__getitem__((j, i))

import numpy as np

class SymmetricArray():
    def __init__(self, input_array):
        self.data = np.asarray(input_array)

    def get_array(self):
        h, w = self.data.shape
        final = np.zeros((h, w))
        for i in range(h):
            for j in range(w):
                if i < j:
                    final[i,j] = self.data[i,j]
                else:
                    final[i,j] = self.data[j,i]
        return final

def BilinearInterpolation(source, point):
    x, y = point
    x1, y1 = int(x) - 1 , int(y) - 1
    x2, y2 = int(x)  , int(y)
    x -= 1
    y -= 1

    Q11 = source[y1][x1]
    Q12 = source[y2][x1]
    Q21 = source[y1][x2]
    Q22 = source[y2][x2]


    R1 = ((x2 - x) / (x2 - x1)) * Q11 + ((x - x1) / (x2 - x1)) * Q21
    R2 = ((x2 - x) / (x2 - x1)) * Q12 + ((x - x1) / (x2 - x1)) * Q22
    P = ((y2 - y) / (y2 - y1)) * R1 + ((y - y1) / (y2 - y1)) * R2

    return  P

def cartesian_product(*vectors):
    if not vectors:
        return [[]]
    
    result = []
    first_vector = vectors[0]
    remaining_vectors = vectors[1:]
    
    for item in first_vector:
        result.extend([[item,] + prod for prod in cartesian_product(*remaining_vectors)])
    
    return result

import numpy as np

def extracting_subpart(input_array, shape, fill_value, position):
    h_tgt, w_tgt = shape
    h, w = input_array.shape
    i, j = position
    
    center_width = min(i, j, (h - i - 1), (w - j - 1))
    
    result = np.full(shape, fill_value)

    if center_width != 0:

        top = max(0, j - center_width)
        bottom = min(h, j + center_width + 1)
        left = max(0, i - center_width)
        right = min(w, i + center_width + 1)
        
        top_offset = max(0, (h_tgt - (bottom - top)) // 2 + 1)
        left_offset = max(0, (w_tgt - (right - left)) // 2 + 1)
        
        subpart = input_array[top:bottom, left:right]
        h_sub, w_sub = subpart.shape
        result[top_offset:top_offset+h_sub, left_offset:left_offset+w_sub] = subpart
        
    return result

        if len(a) == 1:
            return a[0][0]

        if len(a) == 2:

        det = 0
        for j in range(len(a)): # 第一行展开
            co = a[0][j] * ((-1) ** j) * determinant(minor(a, 0, j))
            det += co

def find_consecutive_sequences(N):
    result = []
    for k in range(1, N + 1):
        if (2 * N) % k == 0:
            x = (2 * N // k - k + 1) // 2
            if x > 0 and (2 * x + k - 1) * k // 2 == N:
                result.append(list(range(x, x + k)))
    return result[1:]

29. Password checking¶

A website requires the users to input username and password to register. Write a program to check the validity of password input by users. Following are the criteria for checking the password:

• At least 1 letter between [a-z]

• At least 1 number between [0-9]

• At least 1 letter between [A-Z]

• At least 1 character from [$#@]

• Minimum length of transaction password: 6

• Maximum length of transaction password: 12

Your program should accept a sequence of comma separated passwords and will check them according to the above criteria. Passwords that match the criteria are to be printed, each separated by a comma.

e.g.

If the following passwords are given as input to the program: ABd1234@1,a F1#,2w3E*,2We3345

Then, the output of the program should be: ABd1234@1

In [ ]:
import re
def password_checking(list_input):
    result = []
    n = len(list_input)
    for i in range(n):
        password = list_input[i]
        if len(list_input[i]) >= 6 and len(list_input[i]) <= 12:
            if re.search(r'[a-z]', password) \
                and re.search(r'[0-9]', password) \
                and re.search(r'[A-Z]', password) \
                and re.search(r'[$#@]', password):
                    result.append(password)
            
    return result
